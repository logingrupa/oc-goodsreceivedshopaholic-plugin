<objective>
Build ParseAndPersistOrchestrator (APPLY-06) — the upload-side orchestrator that turns a `.HTM` file's raw bytes into a persisted `Invoice` row + `InvoiceLine` children with match resolutions, all wrapped in ONE DB::transaction. Plus the override-reimport entry point `runOverride()` (APPLY-08 / D-26..D-28) that persists a NEW invoice with `override_of_invoice_id` pointing to the prior — apply later runs ADDITIVELY (per D12 add-on-top semantics).

The orchestrator is the single place the controller calls during upload. It owns the transaction boundary so the controller can stay thin. On any parse / persist / match failure, the entire transaction rolls back atomically and the failure logs through ImportAuditService::logReject AFTER the rollback (so the log line records the failure outcome, not a half-committed state).

Purpose: Per D-22, parse → resolve → dup-check → persist invoice → match → persist lines is a unit. Half a parse with no lines is corruption. Half a match with some lines marked but others not is corruption. The orchestrator's tx boundary guarantees atomicity. Per D-26, override-reimport gets its own method (not a flag) so the controller's opt-in is loud and reviewable.

Output: One orchestrator class + one test file with 5+ cases.
</objective>

       <?php
       declare(strict_types=1);
       namespace Logingrupa\GoodsReceivedShopaholic\Classes\Orchestrator;

       use Carbon\Carbon;
       use Illuminate\Support\Facades\DB;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Dto\MatchedLine;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Dto\ParsedInvoice;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Exception\DuplicateInvoiceException;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Exception\GoodsReceivedException;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Match\EanMatcherService;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Parser\HtmInvoiceParser;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Support\ImportAuditService;
       use Logingrupa\GoodsReceivedShopaholic\Models\Invoice;
       use Logingrupa\GoodsReceivedShopaholic\Models\InvoiceLine;

       /**
        * Upload-side orchestrator (APPLY-06 / APPLY-08 / D-21..D-28).
        *
        * Wraps parse → dup-check → persist invoice → match → persist lines in
        * ONE DB::transaction. Apply (stock writes) happens LATER via plan 03-07
        * ApplyOrchestrator — this orchestrator NEVER touches offer.quantity.
        *
        * On any parse / persist / match exception, the transaction rolls back
        * atomically; the orchestrator then logs a reject event AFTER rollback
        * (logging inside the tx would record nothing committed) and re-throws.
        *
        * runOverride() is the explicit override-reimport entry point per D12
        * add-on-top semantics — caller passes the prior invoice id; the new
        * invoice gets override_of_invoice_id set + a suffixed invoice_number
        * to satisfy the UNIQUE index.
        */
       final class ParseAndPersistOrchestrator
       {
           private const MODE_NORMAL = 'normal';
           private const MODE_OVERRIDE = 'override';

           public function __construct(
               private readonly HtmInvoiceParser $obParser = new HtmInvoiceParser(),
               private readonly EanMatcherService $obMatcher = new EanMatcherService(),
               private readonly ImportAuditService $obAudit = new ImportAuditService(),
           ) {}

           public function run(string $sHtmlContent, string $sSourceFilename, int $iAppliedByUserId): Invoice
           {
               return $this->runWithStrategy($sHtmlContent, $sSourceFilename, $iAppliedByUserId, self::MODE_NORMAL, null);
           }

           public function runOverride(string $sHtmlContent, string $sSourceFilename, int $iPriorInvoiceId, int $iAppliedByUserId): Invoice
           {
               return $this->runWithStrategy($sHtmlContent, $sSourceFilename, $iAppliedByUserId, self::MODE_OVERRIDE, $iPriorInvoiceId);
           }

           private function runWithStrategy(string $sHtml, string $sFilename, int $iUserId, string $sMode, ?int $iPriorInvoiceId): Invoice
           {
               try {
                   /** @var Invoice $obInvoice */
                   $obInvoice = DB::transaction(function () use ($sHtml, $sFilename, $iUserId, $sMode, $iPriorInvoiceId): Invoice {
                       $obParsed = $this->obParser->parse($sHtml, $sFilename);

                       if ($sMode === self::MODE_NORMAL) {
                           $this->assertNotDuplicate($obParsed->invoice_number);
                       }

                       $obInvoice = $this->persistInvoice($obParsed, $iUserId, $sMode, $iPriorInvoiceId);

                       $arMatched = $this->matchAllLines($obParsed);
                       $this->persistLines((int) $obInvoice->id, $arMatched);
                       $this->updateInvoiceCounters($obInvoice, $arMatched);

                       $this->obAudit->logParse((int) $obInvoice->id, $obParsed);
                       return $obInvoice;
                   });
                   return $obInvoice;
               } catch (GoodsReceivedException $obException) {
                   // Plugin-typed exception — log + re-throw. Tx already rolled back.
                   $this->obAudit->logReject(get_class($obException), array_merge(
                       $obException->arContext,
                       ['source_filename' => $sFilename, 'mode' => $sMode],
                   ));
                   throw $obException;
               }
           }

           /** @throws DuplicateInvoiceException */
           private function assertNotDuplicate(string $sInvoiceNumber): void
           {
               $obPrior = Invoice::where('invoice_number', $sInvoiceNumber)->lockForUpdate()->first();
               if ($obPrior === null) { return; }
               if ($obPrior->status !== Invoice::STATUS_APPLIED) { return; }

               throw new DuplicateInvoiceException(
                   (string) \Lang::get('logingrupa.goodsreceivedshopaholic::lang.exception.duplicate_invoice'),
                   [
                       'invoice_number' => $sInvoiceNumber,
                       'prior_invoice_id' => (int) $obPrior->id,
                       'prior_applied_at' => $obPrior->applied_at?->toIso8601String(),
                       'prior_applied_by' => $obPrior->applied_by_user_id,
                   ],
               );
           }

           private function persistInvoice(ParsedInvoice $obParsed, int $iUserId, string $sMode, ?int $iPriorInvoiceId): Invoice
           {
               $sInvoiceNumber = $obParsed->invoice_number;
               $iOverrideOf = null;
               if ($sMode === self::MODE_OVERRIDE && $iPriorInvoiceId !== null) {
                   // Suffix the invoice_number to satisfy UNIQUE index. The prior
                   // invoice keeps the canonical number; the override is a derived
                   // label pointing back via override_of_invoice_id.
                   $sInvoiceNumber = sprintf('%s-OVR-%d', $obParsed->invoice_number, $iPriorInvoiceId);
                   $iOverrideOf = $iPriorInvoiceId;
               }
               return Invoice::create([
                   'invoice_number' => $sInvoiceNumber,
                   'invoice_date' => $obParsed->invoice_date,
                   'country_code' => $obParsed->country_code,
                   'source_filename' => $obParsed->source_filename,
                   'status' => Invoice::STATUS_PARSED,
                   'total_lines' => count($obParsed->lines),
                   'applied_by_user_id' => $iUserId,
                   'parsed_at' => Carbon::now(),
                   'override_of_invoice_id' => $iOverrideOf,
               ]);
           }

           /** @return list<MatchedLine> */
           private function matchAllLines(ParsedInvoice $obParsed): array
           {
               $arEans = array_map(fn ($obLine) => $obLine->ean, $obParsed->lines);
               $arMatchMap = $this->obMatcher->matchBatch($arEans);
               return $this->obMatcher->buildMatchedLines($obParsed->lines, $arMatchMap);
           }

           /** @param list<MatchedLine> $arMatched */
           private function persistLines(int $iInvoiceId, array $arMatched): void
           {
               $sNow = Carbon::now()->toDateTimeString();
               $arRows = [];
               foreach ($arMatched as $obMatched) {
                   $arRows[] = [
                       'invoice_id' => $iInvoiceId,
                       'row_index' => $obMatched->line->row_index,
                       'ean' => $obMatched->line->ean,
                       'product_name_raw' => $obMatched->line->product_name_raw,
                       'qty' => $obMatched->line->qty,
                       'unit_price' => $obMatched->line->unit_price,
                       'matched_offer_id' => $obMatched->matched_offer_id,
                       'match_strategy' => $obMatched->match_strategy,
                       'applied' => false,
                       'created_at' => $sNow,
                       'updated_at' => $sNow,
                   ];
               }
               if ($arRows !== []) {
                   InvoiceLine::insert($arRows);
               }
           }

           /** @param list<MatchedLine> $arMatched */
           private function updateInvoiceCounters(Invoice $obInvoice, array $arMatched): void
           {
               $iMatched = 0;
               $iUnmatched = 0;
               foreach ($arMatched as $obM) {
                   if ($obM->matched_offer_id !== null) { $iMatched++; } else { $iUnmatched++; }
               }
               $obInvoice->matched_lines = $iMatched;
               $obInvoice->unmatched_lines = $iUnmatched;
               $obInvoice->saveQuietly();
           }
       }
