<objective>
Build ApplyOrchestrator (APPLY-07) — the apply-side orchestrator. Locks the Invoice row, validates status, runs StockApplyService + ActiveFlagService + status flip + audit log inside ONE DB::transaction, then flushes caches AFTER the commit. The single load-bearing concurrency boundary in Phase 3.

Plus 6 dedicated QA test files (QA-03 four tests + QA-08 two tests) — each in its own file because the QA spec names them as separate test classes, and isolated files make grep-by-name and CI failure attribution clearer.

Purpose: Per D-23..D-25, the apply path is the place ALL invariants converge:
- Concurrency: lockForUpdate serializes apply on the same invoice across requests
- Idempotency: status check after lock prevents double-apply
- Atomicity: stock writes + active-flag toggles + status flip + audit log either ALL commit or ALL roll back
- Cache consistency: flush AFTER commit so stale data isn't repopulated mid-transaction
- Override-reimport semantics: by NOT special-casing override invoices, ADD-ON-TOP per D12 emerges naturally — applying an override invoice writes qty additively, just like a normal invoice

Output: One orchestrator class + 6 test files. This plan is the largest in Phase 3 (3 tasks, 7 files modified) and closes 4 of the 16 phase requirements.

       <?php
       declare(strict_types=1);
       namespace Logingrupa\GoodsReceivedShopaholic\Classes\Orchestrator;

       use Carbon\Carbon;
       use Illuminate\Support\Facades\DB;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Apply\ActiveFlagService;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Apply\StockApplyOutcome;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Apply\StockApplyService;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Dto\ApplyResult;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Exception\ApplyAlreadyDoneException;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Exception\GoodsReceivedException;
       use Logingrupa\GoodsReceivedShopaholic\Classes\Support\ImportAuditService;
       use Logingrupa\GoodsReceivedShopaholic\Models\Invoice;

       /**
        * Apply-side orchestrator (APPLY-07 / APPLY-08 / QA-03 / QA-08 / D-23..D-25).
        *
        * Sequence:
        *   1) DB::transaction starts (the FIRST transaction is the lock holder)
        *   2) Invoice::where('id', $id)->lockForUpdate()->firstOrFail()
        *   3) status==='applied'? → throw ApplyAlreadyDoneException (with prior result context)
        *   4) StockApplyService::apply → returns StockApplyOutcome (ApplyResult + affected_offer_ids)
        *   5) ActiveFlagService::reconcile($outcome->affected_offer_ids) — SAME tx (QA-08)
        *   6) Invoice.status = 'applied'; applied_at, applied_by_user_id, stock_added_units; saveQuietly
        *   7) ImportAuditService::logApply
        *   8) tx COMMITS
        *   9) StockApplyService::flushAffectedCaches($outcome->affected_offer_ids) — OUTSIDE tx (D-10)
        *
        * Override-reimport semantics (D-28): No special case. The orchestrator
        * runs identically on override invoices. Their override_of_invoice_id
        * pointer is metadata — apply behavior is ADDITIVE because StockApply
        * does qty += delta. Pinned by OverrideReimportAddsOnTopTest.
        *
        * Concurrency contract:
        *   - lockForUpdate() inside DB::transaction holds the row lock until
        *     commit OR rollback. Concurrent apply() calls on the SAME invoice
        *     id are serialized.
        *   - QA-03 LockForUpdateSerializesConcurrentApplyTest pins this via
        *     a structural assertion (lockForUpdate is in the query log BEFORE
        *     the status check fires).
        */
       final class ApplyOrchestrator
       {
           public function __construct(
               private readonly StockApplyService $obStockApply = new StockApplyService(),
               private readonly ActiveFlagService $obActiveFlag = new ActiveFlagService(),
               private readonly ImportAuditService $obAudit = new ImportAuditService(),
           ) {}

           /**
            * @throws ApplyAlreadyDoneException
            * @throws \Illuminate\Database\Eloquent\ModelNotFoundException
            */
           public function apply(int $iInvoiceId, int $iAppliedByUserId): ApplyResult
           {
               /** @var StockApplyOutcome $obOutcome */
               $obOutcome = DB::transaction(function () use ($iInvoiceId, $iAppliedByUserId): StockApplyOutcome {
                   return $this->executeInTransaction($iInvoiceId, $iAppliedByUserId);
               });

               // Post-commit cache flush (D-10).
               $this->obStockApply->flushAffectedCaches($obOutcome->affected_offer_ids);

               return $obOutcome->result;
           }

           /** @throws ApplyAlreadyDoneException */
           private function executeInTransaction(int $iInvoiceId, int $iAppliedByUserId): StockApplyOutcome
           {
               /** @var Invoice $obInvoice */
               $obInvoice = Invoice::where('id', $iInvoiceId)->lockForUpdate()->firstOrFail();

               $this->assertNotApplied($obInvoice);

               /** @var \October\Rain\Database\Collection $obMatchedLines */
               $obMatchedLines = $obInvoice->lines()->whereNotNull('matched_offer_id')->get();

               $obOutcome = $this->obStockApply->apply($obInvoice, $obMatchedLines);

               // ActiveFlagService runs INSIDE the same tx (QA-08).
               $this->obActiveFlag->reconcile($obOutcome->affected_offer_ids);

               $this->markInvoiceApplied($obInvoice, $obOutcome, $iAppliedByUserId);

               $this->obAudit->logApply((int) $obInvoice->id, $obOutcome->result, $iAppliedByUserId);

               return $obOutcome;
           }

           /** @throws ApplyAlreadyDoneException */
           private function assertNotApplied(Invoice $obInvoice): void
           {
               if ($obInvoice->status !== Invoice::STATUS_APPLIED) { return; }

               throw new ApplyAlreadyDoneException(
                   (string) \Lang::get('logingrupa.goodsreceivedshopaholic::lang.exception.apply_already_done'),
                   [
                       'invoice_id' => (int) $obInvoice->id,
                       'invoice_number' => $obInvoice->invoice_number,
                       'prior_applied_at' => $obInvoice->applied_at?->toIso8601String(),
                       'prior_applied_by' => $obInvoice->applied_by_user_id,
                       'prior_stock_added_units' => (int) $obInvoice->stock_added_units,
                   ],
               );
           }

           private function markInvoiceApplied(Invoice $obInvoice, StockApplyOutcome $obOutcome, int $iAppliedByUserId): void
           {
               $obInvoice->status = Invoice::STATUS_APPLIED;
               $obInvoice->applied_at = Carbon::now();
               $obInvoice->applied_by_user_id = $iAppliedByUserId;
               $obInvoice->stock_added_units = $obOutcome->result->units_added;
               $obInvoice->saveQuietly();
           }
       }
