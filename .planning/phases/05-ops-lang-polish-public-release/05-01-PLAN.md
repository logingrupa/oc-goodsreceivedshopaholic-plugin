    function arrayKeysRecursive(array $arInput, string $sPrefix = ''): array {
        $arResult = [];
        foreach ($arInput as $sKey => $mValue) {
            $sFullKey = $sPrefix === '' ? (string)$sKey : $sPrefix . '.' . $sKey;
            if (is_array($mValue)) {
                $arResult = array_merge($arResult, arrayKeysRecursive($mValue, $sFullKey));
                continue;
            }
            $arResult[] = $sFullKey;
        }
        sort($arResult);
        return $arResult;
