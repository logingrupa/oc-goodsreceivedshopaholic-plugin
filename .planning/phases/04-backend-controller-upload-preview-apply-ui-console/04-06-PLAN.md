```php
private const OVERRIDE_LITERAL = 'OVERRIDE';

private const RESET_LITERAL = 'RESET';

/**
 * AJAX handler: render the OVERRIDE warning modal (UI-10 / D-18).
 *
 * @return array<string, string>
 *
 * @throws AjaxException
 */
public function onOverrideShowConfirm(): array
{
    $this->assertPermission('logingrupa.goodsreceived.override_invoices');

    $iPriorInvoiceId = (int) Input::get('prior_invoice_id');
    if ($iPriorInvoiceId <= 0) {
        throw new AjaxException(['message' => 'prior_invoice_id is required.']);
    }

    return [
        '#overrideConfirm' => $this->makePartial('_partials/_override_confirm', [
            'prior_invoice_id' => $iPriorInvoiceId,
        ]),
    ];
}

/**
 * AJAX handler: process the typed-OVERRIDE submission. Validates the
 * literal then routes through ParseAndPersistOrchestrator::runOverride.
 * The new Invoice's `override_of_invoice_id` points at the prior; the
 * operator subsequently clicks Apply on the new row to execute the
 * ADD-ON-TOP write (Phase 3 D-28 — apply orchestrator runs unchanged).
 *
 * @return array<string, string>
 *
 * @throws AjaxException
 */
public function onOverrideConfirm(): array
{
    $this->assertPermission('logingrupa.goodsreceived.override_invoices');

    $sTyped = (string) Input::get('confirm_typed');
    if ($sTyped !== self::OVERRIDE_LITERAL) {
        throw new AjaxException([
            'message' => sprintf('Type %s exactly to confirm override.', self::OVERRIDE_LITERAL),
        ]);
    }

    $iPriorInvoiceId = (int) Input::get('prior_invoice_id');
    if ($iPriorInvoiceId <= 0) {
        throw new AjaxException(['message' => 'prior_invoice_id is required.']);
    }

    $arFiles = Input::file('files');
    if (! is_array($arFiles) || $arFiles === []) {
        throw new AjaxException(['message' => 'No file provided for override.']);
    }
    $obFile = $arFiles[0];
    if (! $obFile instanceof UploadedFile) {
        throw new AjaxException(['message' => 'Invalid uploaded file.']);
    }

    $this->assertHtmFile($obFile);
    $iUserId = $this->resolveBackendUserId();
    $sHtml = (string) file_get_contents((string) $obFile->getRealPath());
    $sFilename = (string) $obFile->getClientOriginalName();

    $obOrchestrator = app(ParseAndPersistOrchestrator::class);
    if (! $obOrchestrator instanceof ParseAndPersistOrchestrator) {
        throw new \RuntimeException('ParseAndPersistOrchestrator failed to resolve.');
    }
    $obInvoice = $obOrchestrator->runOverride($sHtml, $sFilename, $iPriorInvoiceId, $iUserId);

    return [
        '#invoicePreviewWrap' => $this->makePartial('_partials/_preview_lines', [
            'invoices' => [$this->buildPreviewPayload($obInvoice)],
        ]),
        '#invoiceRejectWrap' => '',
    ];
}
```

The override flow re-ingests the bytes; D-21 has the controller calling runOverride directly with the new bytes.

```php
/**
 * AJAX handler: render the RESET warning modal with pre-mutation snapshot
 * counts. Triggered by the initial-reset checkbox on the upload form
 * (UI-08 / D-22, D-23).
 *
 * @return array<string, string>
 *
 * @throws AjaxException
 */
public function onInitialResetShowConfirm(): array
{
    $this->assertPermission('logingrupa.goodsreceived.run_initial_reset');

    if (! SettingsAccessor::allowInitialReset()) {
        throw new AjaxException([
            'message' => (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.exception.initial_reset_not_allowed'),
            'reason'  => 'settings_disabled',
        ]);
    }

    if (Invoice::where('initial_reset_applied', true)->exists()) {
        throw new AjaxException([
            'message' => (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.exception.initial_reset_not_allowed'),
            'reason'  => 'already_applied',
        ]);
    }

    $iOfferCount = (int) \Lovata\Shopaholic\Models\Offer::count();
    $iProductCount = (int) \Lovata\Shopaholic\Models\Product::count();

    return [
        '#initialResetConfirm' => $this->makePartial('_partials/_initial_reset_confirm', [
            'offer_count'   => $iOfferCount,
            'product_count' => $iProductCount,
        ]),
    ];
}

/**
 * AJAX handler: validate the typed RESET literal, then run InitialReset
 * BEFORE ApplyOrchestrator::apply on the uploaded invoice (D-24).
 *
 * @return array<string, string>
 *
 * @throws AjaxException
 */
public function onInitialResetConfirm(): array
{
    $this->assertPermission('logingrupa.goodsreceived.run_initial_reset');

    $sTyped = (string) Input::get('confirm_typed');
    if ($sTyped !== self::RESET_LITERAL) {
        throw new AjaxException([
            'message' => sprintf('Type %s exactly to confirm initial reset.', self::RESET_LITERAL),
        ]);
    }

    $arFiles = Input::file('files');
    if (! is_array($arFiles) || $arFiles === []) {
        throw new AjaxException(['message' => 'No file provided for initial reset apply.']);
    }
    $obFile = $arFiles[0];
    if (! $obFile instanceof UploadedFile) {
        throw new AjaxException(['message' => 'Invalid uploaded file.']);
    }

    $this->assertHtmFile($obFile);
    $iUserId = $this->resolveBackendUserId();
    $sHtml = (string) file_get_contents((string) $obFile->getRealPath());
    $sFilename = (string) $obFile->getClientOriginalName();

    $obParse = app(ParseAndPersistOrchestrator::class);
    $obReset = app(\Logingrupa\GoodsReceivedShopaholic\Classes\Apply\InitialResetService::class);
    $obApply = app(ApplyOrchestrator::class);
    if (! $obParse instanceof ParseAndPersistOrchestrator
        || ! $obReset instanceof \Logingrupa\GoodsReceivedShopaholic\Classes\Apply\InitialResetService
        || ! $obApply instanceof ApplyOrchestrator) {
        throw new \RuntimeException('Initial-reset service stack failed to resolve.');
    }

    $obInvoice = $obParse->run($sHtml, $sFilename, $iUserId);
    $obReset->reset($obInvoice);
    $obResult = $obApply->apply((int) $obInvoice->id, $iUserId);

    return [
        '#applyResult' => $this->makePartial('_partials/_apply_success', [
            'invoice_id' => (int) $obInvoice->id,
            'result'     => $obResult,
        ]),
    ];
}
```

L10 instanceof guards on every `app()` resolution per D-03-07-03.

The order Reset → Apply matches D-24 EXACTLY: reset runs FIRST (zeros all offers + deactivates products + flips Invoice.initial_reset_applied=true), THEN apply increments stock from the new invoice.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| onOverrideConfirm typed-OVERRIDE | operator-typed string verified server-side; client-side check is UX only |
| onInitialResetConfirm typed-RESET | same pattern; server is source of truth |
| InitialResetService.reset | one-shot enforced by Phase 3 D-17; controller adds permission gate + setting gate as defense-in-depth |
| Per-action permission gates | override_invoices + run_initial_reset checked at handler entry |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06-01 | Spoofing | client-side only typed-string check bypassed by curl | mitigate | Server-side strict equality check on the literal string before the orchestrator is even called; AjaxException returned on mismatch |
