```php
private const ALLOWED_EXTENSIONS = ['htm']; // .HTM only per D-06; case-folded comparison

private const MAX_FILE_SIZE_BYTES = 10 * 1024 * 1024; // 10 MB per D-07

private const FILENAME_INVOICE_NUMBER_REGEX = '/^Nr_PRO(\d+)_/i'; // D-16

/**
 * AJAX handler: ingest an array of uploaded .HTM files. For each file:
 *   1) Validate extension + size.
 *   2) Pre-parse duplicate detection on filename invoice_number (UI-09 / D-16).
 *   3) Route through ParseAndPersistOrchestrator::run (which itself does
 *      body-side duplicate detection + persists Invoice + lines in a single tx).
 *   4) Aggregate per-file outcomes into one of: preview / reject / error cards.
 *
 * Response shape:
 *   [
 *     '#invoicePreviewWrap' => $this->makePartial('_partials/_preview_lines', ['invoices' => ...]),
 *     '#invoiceRejectWrap'  => $this->makePartial('_partials/_reject', ['rejected' => ...]),
 *     '#invoiceUploadErrors' => $this->makePartial('_partials/_upload_errors', ['errors' => ...]),
 *   ]
 *
 * @throws AjaxException When permission is missing or no files present.
 * @return array<string, string>
 */
public function onUpload(): array
{
    $this->assertPermission('logingrupa.goodsreceived.upload_invoices');

    $arFiles = Input::file('files');
    if (! is_array($arFiles) || $arFiles === []) {
        throw new AjaxException(['message' => (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.upload.no_files')]);
    }

    $iUserId = $this->resolveBackendUserId();

    $arPreviews = [];
    $arRejects = [];
    $arErrors = [];

    foreach ($arFiles as $obFile) {
        if (! $obFile instanceof UploadedFile) {
            continue;
        }
        try {
            $this->assertHtmFile($obFile);
            $sFilename = $obFile->getClientOriginalName();
            $sNumber = $this->extractInvoiceNumberFromFilename($sFilename);

            // UI-09 / D-16: pre-parse duplicate gate.
            if ($sNumber !== null) {
                $obPrior = Invoice::where('invoice_number', $sNumber)
                    ->where('status', Invoice::STATUS_APPLIED)
                    ->first();
                if ($obPrior instanceof Invoice) {
                    $arRejects[] = $this->buildRejectPayload($obPrior);
                    continue;
                }
            }

            $sHtml = (string) file_get_contents((string) $obFile->getRealPath());
            $obOrchestrator = app(ParseAndPersistOrchestrator::class);
            if (! $obOrchestrator instanceof ParseAndPersistOrchestrator) {
                throw new \RuntimeException('ParseAndPersistOrchestrator failed to resolve.');
            }
            $obInvoice = $obOrchestrator->run($sHtml, $sFilename, $iUserId);
            $arPreviews[] = $this->buildPreviewPayload($obInvoice);
        } catch (GoodsReceivedException $obException) {
            $arErrors[] = [
                'filename' => $obFile->getClientOriginalName(),
                'message'  => $obException->getMessage(),
            ];
        } catch (Throwable $obException) {
            $arErrors[] = [
                'filename' => $obFile->getClientOriginalName(),
                'message'  => $obException->getMessage(),
            ];
        }
    }

    return [
        '#invoicePreviewWrap' => $this->makePartial('_partials/_preview_lines', ['invoices' => $arPreviews]),
        '#invoiceRejectWrap'  => $this->makePartial('_partials/_reject', ['rejects' => $arRejects]),
        '#invoiceUploadErrors' => $this->makePartial('_partials/_upload_errors', ['errors' => $arErrors]),
    ];
}

/**
 * Pre-parse duplicate detection helper — D-16 regex extraction.
 * Returns null when filename does not match the expected pattern; null
 * triggers full parse path (body-side InvoiceNumberResolver fallback owns
 * the actual contract — this gate is an OPTIMIZATION, not the only check).
 */
private function extractInvoiceNumberFromFilename(string $sFilename): ?string
{
    if (preg_match(self::FILENAME_INVOICE_NUMBER_REGEX, $sFilename, $arMatches) !== 1) {
        return null;
    }
    return 'PRO'.$arMatches[1];
}

/**
 * Server-side .htm extension + size whitelist (D-06 / D-07). Throws
 * GoodsReceivedException so the outer foreach catches and reports
 * structurally — same path as parser failures.
 *
 * @throws GoodsReceivedException
 */
private function assertHtmFile(UploadedFile $obFile): void
{
    $sExtension = strtolower((string) $obFile->getClientOriginalExtension());
    if (! in_array($sExtension, self::ALLOWED_EXTENSIONS, true)) {
        throw new \Logingrupa\GoodsReceivedShopaholic\Classes\Exception\MalformedHtmException(
            (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.upload.bad_extension'),
            ['extension' => $sExtension, 'filename' => (string) $obFile->getClientOriginalName()],
        );
    }
    if ((int) $obFile->getSize() > self::MAX_FILE_SIZE_BYTES) {
        throw new \Logingrupa\GoodsReceivedShopaholic\Classes\Exception\MalformedHtmException(
            (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.upload.too_large'),
            ['size' => (int) $obFile->getSize(), 'max' => self::MAX_FILE_SIZE_BYTES],
        );
    }
}

private function assertPermission(string $sPermissionKey): void
{
    if (! BackendAuth::userHasAccess($sPermissionKey)) {
        throw new AjaxException([
            'message' => (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.flash.forbidden'),
        ]);
    }
}

private function resolveBackendUserId(): int
{
    $obUser = BackendAuth::getUser();
    if ($obUser === null) {
        throw new AjaxException([
            'message' => (string) Lang::get('logingrupa.goodsreceivedshopaholic::lang.flash.forbidden'),
        ]);
    }
    return (int) $obUser->id;
}

/** @return array<string, mixed> */
private function buildRejectPayload(Invoice $obPrior): array
{
    return [
        'invoice_number'          => (string) $obPrior->invoice_number,
        'prior_applied_at'        => $obPrior->applied_at?->toIso8601String(),
        'prior_applied_by'        => $obPrior->applied_by_user_id,
        'prior_stock_added_units' => (int) $obPrior->stock_added_units,
        'prior_invoice_id'        => (int) $obPrior->id,
    ];
}

/** @return array<string, mixed> */
private function buildPreviewPayload(Invoice $obInvoice): array
{
    return [
        'invoice'        => $obInvoice,
        'lines'          => InvoiceLine::where('invoice_id', $obInvoice->id)->orderBy('row_index')->get(),
        'total_units'    => (int) InvoiceLine::where('invoice_id', $obInvoice->id)->sum('qty'),
        'matched_count'  => (int) $obInvoice->matched_lines,
        'unmatched_count' => (int) $obInvoice->unmatched_lines,
    ];
}
```

A) Add to `controllers/Invoices.php`:

```php
/**
 * AJAX handler: update override_qty or override_reason on a single
 * InvoiceLine. Triggered by data-track-input bindings on the preview
 * line table (UI-03 / D-09). Returns the updated line as a small JSON
 * payload so the operator's edit is reflected without a full preview
 * re-render.
 *
 * @return array<string, mixed>
 *
 * @throws AjaxException
 */
public function onUpdateLine(): array
{
    $this->assertPermission('logingrupa.goodsreceived.upload_invoices');

    $iLineId = (int) Input::get('line_id');
    if ($iLineId <= 0) {
        throw new AjaxException(['message' => 'line_id is required.']);
    }

    /** @var InvoiceLine|null $obLine */
    $obLine = InvoiceLine::find($iLineId);
    if (! $obLine instanceof InvoiceLine) {
        throw new AjaxException(['message' => sprintf('InvoiceLine #%d not found.', $iLineId)]);
    }

    $bDirty = false;
    if (Input::has('override_qty')) {
        $iOverrideQty = (int) Input::get('override_qty');
        if ($iOverrideQty < 0) {
            throw new AjaxException(['message' => 'override_qty must be non-negative.']);
        }
        $obLine->override_qty = $iOverrideQty;
        $bDirty = true;
    }

    if (Input::has('override_reason')) {
        $sReason = trim((string) Input::get('override_reason'));
        $obLine->override_reason = $sReason !== '' ? $sReason : null;
        $bDirty = true;
    }

    if (! $bDirty) {
        return ['#noop' => ''];
    }

    $obLine->saveQuietly();

    return [
        'line_id'         => $iLineId,
        'override_qty'    => $obLine->override_qty,
        'override_reason' => $obLine->override_reason,
    ];
}
```

`saveQuietly` — same semantics as Phase 3 stock writes; we're updating audit-only override fields and don't want to fire model handlers.

The `if (! $obLine instanceof InvoiceLine)` guard is the L10-friendly pattern for narrowing `Model|null` returns from `Model::find` per D-03-07-03 — no inline @var.
